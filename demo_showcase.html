// ERC-20 代币合约
contract DHDToken {
  string public name = "ADHD Token";
  string public symbol = "DHD";
  uint8 public decimals = 18;
  
  function mint(
    address to,
    uint256 amount
  ) external;
}
Rewarder.sol 
// 奖励分发合约
contract Rewarder {
  DHDToken public token;
  uint256 public cooldown = 30 minutes;
  
  function reward(
    address user,
    uint256 score
  ) external;
  
  // 防滥用机制
  mapping(address => uint256) lastClaim;
}
